// Simple chatbot demo
function chatbotResponse(question) {
  question = question.toLowerCase();
  if (question.includes("skills")) {
    return "I am skilled in HTML, CSS, JavaScript, and Python.";
  } else if (question.includes("projects")) {
    return "Check out my GitHub projects in the Projects section!";
  } else if (question.includes("resume")) {
    return "You can download my resume in the Resume section.";
  } else {
    return "Thanks for visiting my portfolio!";
  }
}

// Example usage in console
console.log(chatbotResponse("Tell me about your skills"));
